	declare K_{{{LapCounter::C}}} LapCounter = {{{LapCounter::C}}}_Get(Page.MainFrame,"lap-counter");
	declare K_{{{RaceTimer::C}}} RaceTimer = {{{RaceTimer::C}}}_Get(Page.MainFrame,"race-timer");

	declare K_{{{CheckpointQuality::C}}} CheckpointQuality = {{{CheckpointQuality::C}}}_Get(Page.MainFrame,"checkpoint-quality");
	declare K_{{{RelayStatus::C}}} RelayStatus = {{{RelayStatus::C}}}_Get(Page.MainFrame,"relay-status");

***Yield***
***

	declare CSmPlayer Owner <=> GetOwner();
	declare K_{{{ChaseTeams::C}}} Team = {{{ChaseTeams::C}}}_GetTeam(Owner);

	{{{Radar::C}}}_Update(Radar,Owner,Relayer);
	{{{RelayStatus::C}}}_Update(RelayStatus,Owner,Relayer,Team.RelayStarted,Team.PrevRelayWasSuccess,Team.CurrentCp,Team.PrevRelayerLogin);
	{{{LapCounter::C}}}_Update(LapCounter,Owner,G_{{{Settings::C}}}.LapCount);
	{{{RaceTimer::C}}}_Update(RaceTimer,Owner);
	LastSuccessCpSpeed = {{{CheckpointQuality::C}}}_Update(CheckpointQuality,Owner,Relayer,PrevRelayer, Team.CurrentCp, Team.LastSuccessRelayCpTime, Team.PrevRelayWasSuccess, LastSuccessCpSpeed);

}

declare Boolean G_Loaded;
Void Unload() {
	G_Loaded = False;
	Layers::Destroy(C);
}

Void Load() {
	if(G_Loaded)return;
	Unload();
	G_Loaded = True;
	Manialink::Load();
	Radar::Load();
	LapCounter::Load();
	RaceTimer::Load();
	CheckpointQuality::Load();
	ChaseTeams::Load();
	Settings::Load();
	RelayStatus::Load();
	Manialink::CreateComponent(C,[CheckpointQuality::C, RelayStatus::C,Radar::C,LapCounter::C,RaceTimer::C,ChaseTeams::C,Settings::C,"GetOwner","GetPlayer"],_Model(),"","","",_Labels());
	Layers::Load();
	Layers::Create(C,Manialink::CreateManialink(C));
	Layers::SetType(C,CUILayer::EUILayerType::Normal);
	Layers::AttachAll(C);
}
