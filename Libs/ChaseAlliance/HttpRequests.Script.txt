Integer GetCurrentPlayerCount(){
	declare Integer PlayerCount = 0;
	foreach (Player in Players) {
		if(!TL::StartsWith("*fakeplayer" , Player.User.Login)){
			PlayerCount+=1;		
		} 
	}
	return PlayerCount;
}
