#Struct K_Settings {
	Integer LapCount;
	Integer RelayerCount;
	Integer RelayGap;
	Integer RelaySortDelay;
}

declare K_Settings G_Settings;

Void ToNet(){
	declare netwrite K_Settings Net_Settings for Teams[0] = G_Settings;
	declare netwrite Integer Net_SettingsChange for Teams[0] = Now;
	Net_Settings = G_Settings;
	Net_SettingsChange = Now;
}

Void SetLapCount(Integer _Value){
	if(G_Settings.LapCount != _Value){
		G_Settings.LapCount = _Value;
		UIManager.UIAll.SendNotice(
			"$"^TL::ColorToText(Colors::C_Warning)^"LapCount changed to "^_Value^" laps", 
			CUIConfig::ENoticeLevel::MatchInfo, Null, CUIConfig::EAvatarVariant::Default, CUIConfig::EUISound::Silence, 0);
		LapCountOverride = _Value;
	}
}

Void SetRelayerCount(Integer _Value){
	if(G_Settings.RelayerCount != _Value){
		G_Settings.RelayerCount = _Value;
		UIManager.UIAll.SendNotice(
			"$"^TL::ColorToText(Colors::C_Warning)^"RelayerCount changed to "^_Value^" relayers", 
			CUIConfig::ENoticeLevel::MatchInfo, Null, CUIConfig::EAvatarVariant::Default, CUIConfig::EUISound::Silence, 0);
		ChaseTeams::SetRelayerCount(_Value);
	}
}

Void SetRelayGap(Integer _Value){
	if(G_Settings.RelayGap != _Value){
		G_Settings.RelayGap = _Value;
		UIManager.UIAll.SendNotice(
			"$"^TL::ColorToText(Colors::C_Warning)^"RelayGap changed to "^_Value^" points",
			CUIConfig::ENoticeLevel::MatchInfo, Null, CUIConfig::EAvatarVariant::Default, CUIConfig::EUISound::Silence, 0);
		ChaseTeams::SetRelayGap(_Value);
	}
}

Void SetRelaySortDelay(Integer _Value){
	if(G_Settings.RelaySortDelay != _Value){
		G_Settings.RelaySortDelay = _Value;
		UIManager.UIAll.SendNotice(
			"$"^TL::ColorToText(Colors::C_Warning)^"RelaySortDelay changed to "^_Value^" ms",
			CUIConfig::ENoticeLevel::MatchInfo, Null, CUIConfig::EAvatarVariant::Default, CUIConfig::EUISound::Silence, 0);
		ChaseTeams::SetRelaySortDelay(_Value);
	}
}

declare Boolean G_Loaded;
Void Unload() {
	G_Loaded = False;
}

Void Load() {
	if(G_Loaded)return;
	Unload();
	G_Loaded = True;
	ApplySettings();
	SetLapCount(5);
	SetRelayerCount(3);
	SetRelayGap(3);
	SetRelaySortDelay(250);
	ToNet();
	ChaseTeams::Load();
}
